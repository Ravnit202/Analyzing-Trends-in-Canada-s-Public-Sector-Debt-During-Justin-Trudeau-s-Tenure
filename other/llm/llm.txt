# Load required libraries
library(rstanarm)
library(tidyverse)
library(arrow)

# Read the data
data <- read_parquet("path_to_your_data_file.parquet")

# Model for percent values
model_percent <- stan_glm(value_percent ~ ref_date + geo + level_of_study + type_of_debt_source,
                          data = data,
                          family = gaussian(link = "identity"),
                          prior = normal(0, 5),
                          chains = 4,
                          iter = 2000)

# Save the model for percent values

                           chains = 4,
                           iter = 2000)

# Save the model for dollar values
